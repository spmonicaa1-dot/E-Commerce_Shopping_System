CREATE TABLE ESHOP_TB(
ORDERID VARCHAR2(12) PRIMARY KEY,
CUSTOMERNAME VARCHAR2(50) NOT NULL,
PRODUCTNAME VARCHAR2(50) NOT NULL,
ORDER_DATE DATE NOT NULL,
QUANTITY NUMBER,
PRICE NUMBER,
REMARKS VARCHAR2(100));


CREATE SEQUENCE ESHOP_SEQ
START WITH 10
MAXVALUE 99
INCREMENT BY 1;
